struct node
{
int data;
struct node *next;
};
typedef struct node node;

node *top;

void push(int value)
{
node *tmp;
tmp = malloc(sizeof(node));
tmp -> data = value;
tmp -> next = top;
top = tmp;
}

int pop()
{
if (top==NULL)
{
printf("STACK IS EMPTY!!!\n");
}
else
{
node *tmp;
int n;
tmp = top;
n = tmp->data;
top = top->next;
printf("\nThe %d number is popped from stack\n",n);
free(tmp);
}
}
void display(node *head)
{
while(head!=NULL)
{
printf("%d\n",head->data);
head=head->next;
}
}
