# define MAXIMUM 10
struct stack
{
int stack_arr[MAXIMUM];
int top;
};
struct stack* stk;


void push(int element)
{
if (stk->top==(MAXIMUM-1))
{
printf("STACK OVERFLOW!!!!!");
}
else
{
stk->top+=1;
stk->stack_arr[stk->top]=element;


}

}
void display()
{
int i;
printf("current stack elements\n");
for (i=stk->top;i>=0;i--)
{
printf("%d\n",stk->stack_arr[i]);
}

}
void pop()
{
int popped_num;
if (stk->top==-1)
printf("STACK IS EMPTY!!!\n");
else
{
popped_num=stk->stack_arr[stk->top];
printf("Number %d successfully popped\n",popped_num);
stk->top-=1;
}
}
void main()
{

printf("ROLL NO : 22318\n");
stk->top=-1;
int choice=1;
int choice1;
do
{
printf("Enter choice :\n 1. ADD ELEMENT \n 2. DELETE ELEMENT \n 3. DISPLAY STACK\n");
scanf("%d",&choice1);

switch (choice1)
{
case 1:
push(3);
break;
case 2:
pop();
break;
case 3:
display();
break;
default:
printf("INVALID INPUT\n");
break;
}
printf("\n PRESS 1 TO CONTINUE.......\n PRESS 0 TO EXIT.........\n");
scanf("%d", &choice);
}while (choice==1);




}
